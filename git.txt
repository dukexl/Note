1.新建一个仓库
2.git clone 到本地
3.git add 文件
4.git status -s 查看状态
5.git push  提交上去
6.git pull  拉去下来

git rm -r note
git commit -m"delete note"
git push -u origin master


删除github 某个文件夹
$ git --help                        # 帮助命令 
$ git pull origin master            # 将远程仓库里面的项目拉下来
$ dir                               # 查看有哪些文件夹
$ git rm -r --cached target         # 删除target文件夹
$ git commit -m '删除了target'      # 提交,添加操作说明
$ git push -u origin master         # 将本次更改更新到github项目上去
